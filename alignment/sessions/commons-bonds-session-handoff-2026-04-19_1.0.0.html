Python processing script template (reuse for Batch 2)

"""Decode base64 HTML blobs, convert to Markdown via BeautifulSoup + markdownify."""
import base64, os
from bs4 import BeautifulSoup
from markdownify import markdownify as md

OUT = "/mnt/user-data/outputs/commons-bonds"
RAW = "/home/claude/commons-bonds-raw"
os.makedirs(OUT, exist_ok=True)

def clean_html(html):
    soup = BeautifulSoup(html, "html.parser")
    for tag in soup(["style", "script"]):
        tag.decompose()
    return str(soup)

# Batch 2 names — replace with actual downloaded files:
MD_NAMES = ["06_chapter_10_and_sweep", "07_chapter_8_two_book_trim",
            "09_old_material_rigor_check", "10_project_separation_folder_reorg"]
HTML_NAMES = ["08_technical_appendix"]   # keeps math

for name in MD_NAMES + HTML_NAMES:
    with open(f"{RAW}/{name}.b64") as f:
        html = base64.b64decode(f.read().strip()).decode("utf-8")
    if name in HTML_NAMES:
        with open(f"{OUT}/{name}.html", "w") as f:
            f.write(html)
        continue
    markdown = md(clean_html(html), heading_style="ATX", bullets="-",
                  strip=["style", "script"])
    # Collapse >2 blank lines
    out_lines, blank = [], 0
    for ln in markdown.split("\n"):
        if ln.strip() == "":
            blank += 1
            if blank <= 2: out_lines.append(ln)
        else:
            blank = 0; out_lines.append(ln)
    with open(f"{OUT}/{name}.md", "w") as f:
        f.write("\n".join(out_lines).strip() + "\n")
