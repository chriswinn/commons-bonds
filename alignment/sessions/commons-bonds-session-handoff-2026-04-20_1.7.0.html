Conversion pipeline (proven)

cat /mnt/user-data/uploads/{filename}.html | python3 -c "
import sys, re
from bs4 import BeautifulSoup
from markdownify import markdownify

html = sys.stdin.read()
soup = BeautifulSoup(html, 'html.parser')
for tag in soup.find_all(['style', 'script']):
    tag.decompose()
md = markdownify(str(soup), heading_style='ATX', bullets='-')
md = re.sub(r'\n{3,}', '\n\n\n', md).strip() + '\n'
print(md)
" > /mnt/user-data/outputs/{filename}.md
